Bio-distribución (Optimizada)

Dispositivo de síntesis de conocimiento (FAIR/CARE / OJS)

La visualización de la distribución de especies de la vida silvestre, mediante mapas, ayuda en la interpretación multidisciplinar de la consulta.

Tip💡 Carga Instantánea

Esta versión optimizada corre de forma nativa en el navegador sin esperas de inicialización de entornos de programación complejos.

Código 
viewof especies_input = Inputs.textarea({
  label: "Especies (una por línea):",
  value: "Panthera onca\nPuma concolor",
  rows: 4
})

viewof max_reg = Inputs.range([10, 200], {
  label: "Registros máx. por especie:", 
  step: 10, 
  value: 50
})
Código 
lista_especies = especies_input.split("\n").map(d => d.trim()).filter(d => d !== "")

// Función asíncrona que consulta los puntos geográficos nativos
async function obtenerOcurrencias(especie, limite) {
  if (!especie) return [];
  const url = `https://gbif.org{encodeURIComponent(especie)}&hasCoordinate=true&limit=${limite}`;
  try {
    const respuesta = await fetch(url);
    const datos = await respuesta.json();
    return (datos.results || []).map(d => ({
      lat: d.decimalLatitude,
      lng: d.decimalLongitude,
      scientificName: d.scientificName,
      datasetKey: d.datasetKey,
      especie_consultada: especie
    }));
  } catch (error) {
    console.error("Error consultando GBIF para: " + especie, error);
    return [];
  }
}

// Unión de promesas para resolver todas las especies en paralelo de forma segura
datos_mapa = Promise.all(lista_especies.map(sp => obtenerOcurrencias(sp, max_reg))).then(res => res.flat())

Mapa de Distribución Geográfica

Código 
container = html`<div style="height:450px; width:100%; border-radius: 8px; border: 1px solid #ddd;"></div>`

// Bloque de lógica cartográfica reactiva encapsulada
{
  // Inicialización del mapa centrada globalmente
  const map = L.map(container).setView([0, 0], 2);
  
  L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {
    attribution: '© OpenStreetMap contributors'
  }).addTo(map);

  // Paleta de colores dinámica para diferenciar especies visualmente
  const especiesUnicas = [...new Set(datos_mapa.map(d => d.especie_consultada))];
  const colores = ["#3b82f6", "#ef4444", "#10b981", "#f59e0b", "#8b5cf6", "#ec4899"];

  // Inyección de marcadores circulares con popup descriptivo
  datos_mapa.forEach(pt => {
    if(pt.lat && pt.lng) {
      const idx = especiesUnicas.indexOf(pt.especie_consultada) % colores.length;
      L.circleMarker([pt.lat, pt.lng], {
        color: colores[idx],
        fillColor: colores[idx],
        radius: 5,
        weight: 1,
        fillOpacity: 0.6
      }).addTo(map).bindPopup(`<b>${pt.scientificName}</b><br>Buscado como: <i>${pt.especie_consultada}</i>`);
    }
  });
  
  // Ajustar el mapa automáticamente si hay datos cargados
  const puntosValidos = datos_mapa.filter(pt => pt.lat && pt.lng);
  if (puntosValidos.length > 0) {
    const bounds = L.latLngBounds(puntosValidos.map(pt => [pt.lat, pt.lng]));
    map.fitBounds(bounds, { padding: [20, 20] });
  }
}

function convertirTablaCSV(data) {
  const encabezado = "decimalLatitude,decimalLongitude,scientificName,datasetKey,especie_consultada\n";
  const filas = data.map(d => `${d.lat},${d.lng},"${d.scientificName}","${d.datasetKey}","${d.especie_consultada}"`).join("\n");
  return encabezado + filas;
}
