  )
}

gbif_name_match <- function(name) {
  url <- paste0("https://api.gbif.org/v1/species/match?name=", URLencode(name))
  tryCatch(fromJSON(url), error = function(e) NULL)
}

# ==========================================
# HELPERS: GLOBI Y PARSER ADAPTADO
# ==========================================
globi_fields <- c("source_taxon_name", "target_taxon_name", "interaction_type",
                   "study_citation", "study_source_citation", "study_external_id")

globi_url <- function(source_taxon, target_taxon) {
  base <- "https://api.globalbioticinteractions.org/interaction"
  field_params <- paste0("field=", sapply(globi_fields, URLencode), collapse = "&")
  paste0(base, "?sourceTaxon=", URLencode(source_taxon),
         "&targetTaxon=", URLencode(target_taxon),
         "&", field_params, "&includeObservations=true&type=json.v2&limit=50")
}

globi_parse <- function(res) {
  tryCatch({
    if (is.null(res)) return(tibble())
    df <- res
    if (is.list(df) && !is.data.frame(df) && !is.null(df$data) && !is.null(df$columns)) {
      mat <- df$data
      cols <- df$columns
      df <- as.data.frame(mat, stringsAsFactors = FALSE)
      colnames(df) <- cols
    }
    if (nrow(df) == 0) return(tibble())
    return(as_tibble(df))
  }, error = function(e) tibble())
}

  evidencia_globi <- tibble()
  tryCatch({
    res_directo <- fromJSON(globi_url(especie1, especie2))
    df_dir <- globi_parse(res_directo)
    
    res_inverso <- fromJSON(globi_url(especie2, especie1))
    df_inv <- globi_parse(res_inverso)
    
    globi_comb <- bind_rows(df_dir, df_inv)
    
    if (nrow(globi_comb) > 0) {
      evidencia_globi <- globi_comb %>%
        mutate(
          # Captura DOIs incrustados en las referencias de texto de GloBI

          doi = tolower(stringr::str_trim(doi)),
          titulo = coalesce(study_citation, "Interacción registrada en GloBI"),
          fuente = "GloBI",
          citas = 0

        select(doi, titulo, fuente, citas)
    }
  }, error = function(e) { NULL })
