Interacciones Bióticas

Ficha FAIR/CARE de interacción entre especies (WebAssembly)

⏳ Primera carga: el entorno estadístico (R) se descarga y arranca completo dentro de tu navegador — puede tardar entre 10 y 30 segundos la primera vez.

Tip💡 Cómo usar esta ficha
  1. Escribí el nombre científico de la Especie A y la Especie B, y verificalos con el botón de chequeo.
  2. (Opcional) Marcá conceptos de los 4 ejes disponibles (sector e interacción ecosistémica, ecología evolutiva, historia de vida/desempeño, aplicaciones) — funcionan como palabras clave que enfocan una búsqueda adicional en un aspecto específico de la interacción.
  3. Apretá “Mapear Interacción”: la app busca primero evidencia general (GloBI, y si no hay, Crossref) y, si marcaste rasgos, agrega una segunda búsqueda enfocada en esos aspectos.
  4. Descargá las citas en .csv si las necesitás para tu propio gestor bibliográfico o como semilla de un proyecto/dataset.
Importante🔎 ¿No estás seguro del nombre científico exacto?
  • GBIF Species Search
  • iNaturalist
  • Tropicos.org (plantas)
  • Catalogue of Life
Advertencia⚠️ Sobre los límites de esta búsqueda

GloBI solo contiene interacciones que alguien formalizó como dataset estructurado — un trabajo clásico o de referencia sobre un sistema puede no estar ahí simplemente porque nadie hizo esa extracción. Crossref busca por coincidencia de texto, no por importancia del trabajo; ordena por citas recibidas cuando puede, pero eso no reemplaza el conocimiento de quien ya conoce la literatura del sistema. Si trabajás con un par de especies en particular, tu propio conocimiento sigue siendo la fuente primaria — esta ficha es un punto de partida y una forma de documentar la búsqueda, no un reemplazo de la revisión experta.

A continuación, la herramienta interactiva de mapeo de interacciones, ejecutada mediante WebAssembly directamente en su navegador:

  )

gbif_name_match <- function(name) {
  url <- paste0("https://api.gbif.org/v1/species/match?name=", URLencode(name))
  tryCatch(fromJSON(url), error = function(e) NULL)
}

# ==========================================
# HELPERS: GloBI — evidencia directa de interacción documentada
# ==========================================
globi_fields <- c("source_taxon_name", "target_taxon_name", "interaction_type",
                   "study_citation", "study_source_citation", "study_external_id")

globi_url <- function(source_taxon, target_taxon) {
  base <- "https://api.globalbioticinteractions.org/interaction"
  field_params <- paste0("field=", sapply(globi_fields, URLencode), collapse = "&")
  paste0(base, "?sourceTaxon=", URLencode(source_taxon),
         "&targetTaxon=", URLencode(target_taxon),
         "&", field_params, "&includeObservations=true&type=json.v2&limit=25")
}

globi_parse <- function(res) {
  tryCatch({
    if (is.null(res)) return(NULL)
    df <- res
    if (is.list(df) && !is.data.frame(df) && !is.null(df$data) && !is.null(df$columns)) {
      mat <- df$data
      df <- as.data.frame(mat, stringsAsFactors = FALSE)
      names(df) <- df$columns
    }
    if (!is.data.frame(df) || nrow(df) == 0) return(NULL)
    df
  }, error = function(e) NULL)
}

globi_buscar <- function(especie_a, especie_b) {
  t0 <- Sys.time()
  r1 <- tryCatch(fromJSON(globi_url(especie_a, especie_b)), error = function(e) NULL)
  d1 <- globi_parse(r1)
  message(paste0("[timing] GloBI dirección 1: ", round(as.numeric(Sys.time() - t0, units = "secs"), 1), "s, ",
                  ifelse(is.null(d1), 0, nrow(d1)), " registros"))

  # Optimización: si la dirección 1 ya trajo evidencia, no gastamos la segunda llamada de red.
  # Trade-off consciente: puede perderse algún registro adicional en la dirección inversa,
  # pero evita duplicar el costo de red cuando ya hay evidencia suficiente.
  if (!is.null(d1) && nrow(d1) > 0) return(distinct(d1))

  t1 <- Sys.time()
  r2 <- tryCatch(fromJSON(globi_url(especie_b, especie_a)), error = function(e) NULL)
  d2 <- globi_parse(r2)
  message(paste0("[timing] GloBI dirección 2: ", round(as.numeric(Sys.time() - t1, units = "secs"), 1), "s, ",
                  ifelse(is.null(d2), 0, nrow(d2)), " registros"))

  if (is.null(d2)) return(NULL)
  distinct(d2)
}

# Convierte study_external_id en un link usable (DOI o URL directa)
formatear_enlace_externo <- function(id) {
  if (is.null(id) || is.na(id) || !nzchar(id)) return(NA)
  id <- trimws(id)
  if (grepl("^https?://", id, ignore.case = TRUE)) return(id)
  if (grepl("^doi:", id, ignore.case = TRUE)) return(paste0("https://doi.org/", sub("^doi:", "", id, ignore.case = TRUE)))
  if (grepl("^10\\.", id)) return(paste0("https://doi.org/", id))
  id  # devolvemos el identificador tal cual si no reconocemos el formato
}

formatear_globi <- function(df) {
  if (is.null(df) || nrow(df) == 0) return(character(0))
  vapply(seq_len(nrow(df)), function(i) {
    tryCatch({
      src <- df$source_taxon_name[i]
      tgt <- df$target_taxon_name[i]
      tipo <- df$interaction_type[i]
      cita <- df$study_citation[i]
      fuente_alt <- df$study_source_citation[i]
      enlace <- tryCatch(formatear_enlace_externo(df$study_external_id[i]), error = function(e) NA)

      cita_final <- if (!is.null(cita) && !is.na(cita) && nzchar(cita)) {
        cita
      } else if (!is.null(fuente_alt) && !is.na(fuente_alt) && nzchar(fuente_alt)) {
        fuente_alt
      } else {
        "(cita no disponible en este registro)"
      }

      # Si GloBI no trae un enlace directo (study_external_id), el dataset de origen
      # simplemente no lo incluyó. Como red de contención, ofrecemos una búsqueda en
      # Google Scholar armada con el texto de la cita — no es un enlace directo garantizado,
      # por eso queda etiquetado explícitamente como "buscar", no como "fuente".
      enlace_txt <- if (!is.null(enlace) && !is.na(enlace)) {
        paste0(" — <a href='", enlace, "' target='_blank'>🔗 fuente</a>")
      } else {
        scholar_url <- paste0("https://scholar.google.com/scholar?q=", URLencode(cita_final))
        paste0(" — <a href='", scholar_url, "' target='_blank'>🔍 buscar cita</a>")
      }

      paste0(src, " — ", tipo, " — ", tgt, ". Fuente: ", cita_final, enlace_txt)
    }, error = function(e) "(No se pudo formatear este registro de GloBI.)")
  }, character(1))
}

# ==========================================
# HELPERS: Crossref — red de seguridad y búsqueda enfocada por rasgos
# mailto: identidad de contacto del proyecto (no personal)
# ==========================================
crossref_url <- function(query, rows = 15) {
  paste0("https://api.crossref.org/works?query=", URLencode(query),
         "&rows=", rows, "&mailto=interdomestic@proton.me")
}

crossref_buscar <- function(query, rows = 15) {
  tryCatch(fromJSON(crossref_url(query, rows))$message$items, error = function(e) NULL)
}

# Ordena los resultados por cantidad de citas recibidas (según el propio Crossref),
# como aproximación imperfecta pero real a "qué tan importante" es un trabajo dentro
# de lo que el motor efectivamente encontró — no soluciona lo que no está indexado.
ordenar_por_citas <- function(items) {
  if (is.null(items) || nrow(items) == 0) return(items)
  citas <- tryCatch(items$`is-referenced-by-count`, error = function(e) NULL)
  if (is.null(citas)) return(items)
  citas[is.na(citas)] <- 0
  items[order(-citas), , drop = FALSE]
}

especies_en_texto <- function(items, especie_a, especie_b) {
  if (is.null(items) || nrow(items) == 0) {
    return(list(ambas = logical(0), alguna = logical(0)))
  }
  texto <- vapply(seq_len(nrow(items)), function(i) {
    titulo <- tryCatch(paste(items$title[[i]], collapse = " "), error = function(e) "")
    abst <- tryCatch(items$abstract[i], error = function(e) NA)
    abst <- if (is.null(abst) || is.na(abst)) "" else abst
    subj <- tryCatch(paste(items$subject[[i]], collapse = " "), error = function(e) "")
    tolower(paste(titulo, abst, subj))
  }, character(1))
  menciona_a <- grepl(tolower(especie_a), texto, fixed = TRUE)
  menciona_b <- grepl(tolower(especie_b), texto, fixed = TRUE)
  list(ambas = menciona_a & menciona_b, alguna = menciona_a | menciona_b)
}

formatear_citas_crossref <- function(items) {
  if (is.null(items) || nrow(items) == 0) return(character(0))
  vapply(seq_len(nrow(items)), function(i) {
    tryCatch({
      titulo <- items$title[[i]]
      titulo <- if (length(titulo) > 0) titulo[1] else "Sin título"
      au <- items$author[[i]]
      autores <- if (is.null(au) || nrow(au) == 0) {
        "Autor/es no disponible/s"
      } else {
        nombres <- paste(au$family, au$given, sep = ", ")
        if (length(nombres) > 3) paste0(paste(nombres[1:3], collapse = "; "), " et al.")
        else paste(nombres, collapse = "; ")
      }
      anio <- tryCatch(items$published[[i]]$`date-parts`[[1]][1], error = function(e) NA)
      revista <- tryCatch(items$`container-title`[[i]][1], error = function(e) NA)
      doi <- items$DOI[i]
      doi_url <- if (!is.null(doi) && !is.na(doi)) paste0("https://doi.org/", doi) else NA
      paste0(
        autores, " (", ifelse(is.na(anio), "s.f.", anio), "). ", titulo,
        ifelse(!is.na(revista), paste0(". ", revista, "."), "."),
        ifelse(!is.na(doi_url), paste0(" ", doi_url), "")
      )
    }, error = function(e) "(No se pudo formatear esta cita.)")
  }, character(1))
}

# Búsqueda general: GloBI y Crossref se combinan siempre, cada cita marcada con su origen.
buscar_evidencia_general <- function(especie_a, especie_b) {
  globi_df <- globi_buscar(especie_a, especie_b)
  citas_globi <- if (!is.null(globi_df) && nrow(globi_df) > 0) {
    paste0("<strong>[GloBI]</strong> ", formatear_globi(globi_df))
  } else character(0)

  q_interaccion <- paste(especie_a, especie_b,
                          "interaction OR predation OR competition OR mutualism OR parasitism OR facilitation")
  items <- ordenar_por_citas(crossref_buscar(q_interaccion, rows = 20))
  f <- especies_en_texto(items, especie_a, especie_b)

  citas_crossref <- character(0)
  nivel_cr <- NA
  consulta_cr <- q_interaccion
  if (!is.null(items) && sum(f$ambas) > 0) {
    citas_crossref <- paste0("<strong>[Crossref]</strong> ", formatear_citas_crossref(items[f$ambas, , drop = FALSE]))
    nivel_cr <- "ambas especies mencionadas"
  } else if (!is.null(items) && sum(f$alguna) > 0) {
    citas_crossref <- paste0("<strong>[Crossref]</strong> ", formatear_citas_crossref(items[f$alguna, , drop = FALSE]))
    nivel_cr <- "al menos una especie mencionada"
  }

  # Si no apareció nada por ninguna vía, una última red: consulta amplia sin palabras de interacción.
  if (length(citas_globi) == 0 && length(citas_crossref) == 0) {
    q_simple <- paste(especie_a, especie_b)
    items2 <- ordenar_por_citas(crossref_buscar(q_simple, rows = 20))
    f2 <- especies_en_texto(items2, especie_a, especie_b)
    if (!is.null(items2) && sum(f2$alguna) > 0) {
      citas_crossref <- paste0("<strong>[Crossref]</strong> ", formatear_citas_crossref(items2[f2$alguna, , drop = FALSE]))
      nivel_cr <- "baja confianza: mención de al menos una especie (consulta amplia)"
      consulta_cr <- q_simple
    }
  }

  citas <- c(citas_globi, citas_crossref)
  query_origen <- c(
    rep("general (GloBI)", length(citas_globi)),
    rep(paste0("general (Crossref: ", ifelse(is.na(nivel_cr), "sin resultados", nivel_cr), ")"), length(citas_crossref))
  )

  fuente <- if (length(citas_globi) > 0 && length(citas_crossref) > 0) {
    "GloBI + Crossref (combinado)"
  } else if (length(citas_globi) > 0) {
    "GloBI"
  } else if (length(citas_crossref) > 0) {
    "Crossref"
  } else {
    "GloBI + Crossref"
  }

  nivel <- if (length(citas_globi) > 0 && !is.na(nivel_cr)) {
    paste0("GloBI: registro directo · Crossref: ", nivel_cr)
  } else if (length(citas_globi) > 0) {
    "registro directo de interacción documentada (GloBI); Crossref no sumó resultados adicionales"
  } else if (!is.na(nivel_cr)) {
    nivel_cr
  } else {
    "sin resultados que mencionen a ninguna de las dos especies, ni en GloBI ni en Crossref"
  }

  list(
    fuente = fuente, nivel = nivel, citas = citas, query_origen = query_origen,
    consulta = paste0("GloBI: ", especie_a, " <-> ", especie_b, " | Crossref: ", consulta_cr)
  )
}

# Búsqueda enfocada: una consulta INDEPENDIENTE por cada concepto marcado,
# etiquetando cada cita con su query_origen (convención "q + concepto", alineada al codebook).
# Nota de rendimiento: cada concepto marcado suma una llamada de red completa —
# con muchos conceptos marcados a la vez, la búsqueda tarda proporcionalmente más.
buscar_evidencia_enfocada <- function(especie_a, especie_b, rasgos_ids) {
  if (length(rasgos_ids) == 0) return(NULL)

  resultados_por_concepto <- lapply(rasgos_ids, function(id) {
    concepto <- items_planos[[id]]
    q <- paste(especie_a, especie_b, concepto$keyword)
    items <- ordenar_por_citas(crossref_buscar(q))
    f <- especies_en_texto(items, especie_a, especie_b)
    if (!is.null(items) && sum(f$alguna) > 0) {
      texto_citas <- formatear_citas_crossref(items[f$alguna, , drop = FALSE])
      data.frame(
        query_origen = paste0("q + ", concepto$label),
        cita = paste0("<em>[q + ", concepto$label, "]</em> ", texto_citas),
        stringsAsFactors = FALSE
      )
    } else {
      NULL
    }
  })

  tabla <- bind_rows(resultados_por_concepto)

  list(
    rasgos = sapply(items_planos[rasgos_ids], function(x) x$label),
    consulta = paste0("Consultas independientes: ",
                       paste(paste0("q + ", sapply(items_planos[rasgos_ids], function(x) x$label)), collapse = ", ")),
    citas = if (is.null(tabla)) character(0) else tabla$cita,
    query_origen = if (is.null(tabla)) character(0) else tabla$query_origen
  )
}

      tags$a(href = "https://posit-dev.github.io/r-shinylive/", target = "_blank", class = "btn btn-sm btn-outline-secondary", "shinylive / webR"),
      tags$a(href = "https://quarto.org", target = "_blank", class = "btn btn-sm btn-outline-secondary", "Quarto")
    )
  )
)

# ==========================================
# SERVIDOR
# ==========================================
server <- function(input, output, session) {

  validacion <- eventReactive(input$verificar, {
    list(
      a = list(consulta = input$especie_a, match = gbif_name_match(input$especie_a)),
      b = list(consulta = input$especie_b, match = gbif_name_match(input$especie_b))
    )
  })

  output$validacion_nombres <- renderUI({
    v <- validacion()
    tagList(lapply(list(v$a, v$b), function(r) {
      m <- r$match
      if (is.null(m) || is.null(m$matchType) || m$matchType == "NONE") {
        div(class = "alert alert-danger", style = "padding:6px; font-size:0.85em; margin-top:4px;",
            paste0("❌ '", r$consulta, "' no se encontró en el backbone de GBIF."))
      } else if (m$matchType == "EXACT") {
        div(class = "alert alert-success", style = "padding:6px; font-size:0.85em; margin-top:4px;",
            paste0("✅ '", r$consulta, "' reconocido como ", m$scientificName, " (", m$status, ")"))
      } else {
        div(class = "alert alert-warning", style = "padding:6px; font-size:0.85em; margin-top:4px;",
            paste0("⚠️ '", r$consulta, "' es una coincidencia aproximada. GBIF sugiere: ", m$canonicalName, "."))
      }
    }))
  })

  rasgos_elegidos_r <- reactive({
    c(input$cb_sector, input$cb_eco_evolutiva, input$cb_historia_vida, input$cb_aplicado)
  })

  output$aviso_conceptos <- renderUI({
    n <- length(rasgos_elegidos_r())
    if (n == 0) return(NULL)
    if (n > 5) {
      div(class = "alert alert-warning", style = "padding: 6px; font-size: 0.8em; margin-top: 6px;",
          paste0("⚠️ Marcaste ", n, " conceptos. Cada uno dispara su propia consulta a Crossref (para poder identificar qué palabra clave trajo cada cita) — con muchos marcados a la vez, la búsqueda enfocada va a tardar bastante más."))
    } else {
      div(style = "font-size: 0.78em; color: #64748b; margin-top: 4px;",
          paste0(n, " concepto(s) marcado(s)."))
    }
  })

  resultado <- eventReactive(input$procesar, {
    req(input$especie_a, input$especie_b)
    t_inicio <- Sys.time()
    withProgress(message = "Consultando GloBI...", value = 0.1, {
      general <- buscar_evidencia_general(input$especie_a, input$especie_b)
      message(paste0("[timing] Evidencia general completa: ",
                      round(as.numeric(Sys.time() - t_inicio, units = "secs"), 1), "s (fuente: ", general$fuente, ")"))
      incProgress(0.5, detail = "Evidencia general lista")

      t_enfocada <- Sys.time()
      enfocada <- buscar_evidencia_enfocada(input$especie_a, input$especie_b, rasgos_elegidos_r())
      message(paste0("[timing] Búsqueda enfocada: ",
                      round(as.numeric(Sys.time() - t_enfocada, units = "secs"), 1), "s"))
      incProgress(0.3, detail = "Listo")

      message(paste0("[timing] TOTAL: ", round(as.numeric(Sys.time() - t_inicio, units = "secs"), 1), "s"))
      list(general = general, enfocada = enfocada)
    })
  })

  output$ficha_resultado <- renderUI({
    res <- resultado()
    req(res)
    g <- res$general
    e <- res$enfocada

    wellPanel(
      style = "background: #ffffff; border: 1px solid #d1d5db;",
      tags$h4("Ficha de Interacción"),
      tags$p(tags$strong("Especie A: "), input$especie_a, tags$br(),
             tags$strong("Especie B: "), input$especie_b),
      tags$hr(),
      tags$h5("Evidencia general"),
      tags$pre(
        paste0(
          "--- METADATA PROFILE (FAIR/CARE) ---\n",
          "Fuente: ", g$fuente, "\n",
          "Consulta: ", g$consulta, "\n",
          "Nivel: ", g$nivel, "\n",
          "Citas encontradas: ", length(g$citas), "\n",
          "Fecha de consulta: ", format(Sys.time(), "%Y-%m-%d %H:%M %Z")
        )
      ),
      if (length(g$citas) == 0) {
        tags$p("No se encontraron citas que mencionen a ninguna de las dos especies.")
      } else {
        tags$ol(lapply(g$citas, function(c) tags$li(HTML(c))))
      },
      if (!is.null(e)) tagList(
        tags$hr(),
        tags$h5(paste0("Evidencia enfocada en: ", paste(e$rasgos, collapse = ", "))),
        tags$p(style = "font-size: 0.8em; color: #64748b;", paste0("Consulta: ", e$consulta)),
        if (length(e$citas) == 0) {
          tags$p("No se encontraron citas enfocadas en estos rasgos para este par de especies.")
        } else {
          tags$ol(lapply(e$citas, function(c) tags$li(HTML(c))))
        }
      ),
      br(),
      downloadButton("descargar_citas", "Descargar todas las citas (.csv)", class = "btn-outline-secondary")
    )
  })

  output$descargar_citas <- downloadHandler(
    filename = function() paste0("interdomestic_citas_", Sys.Date(), ".csv"),
    content = function(file) {
      res <- resultado()
      req(res)
      g <- res$general
      e <- res$enfocada

      filas_generales <- if (length(g$citas) > 0) {
        data.frame(especie_a = input$especie_a, especie_b = input$especie_b,
                   bloque = "general", fuente = g$fuente, query_origen = g$query_origen,
                   cita = g$citas, stringsAsFactors = FALSE)
      } else NULL

      filas_enfocadas <- if (!is.null(e) && length(e$citas) > 0) {
        data.frame(especie_a = input$especie_a, especie_b = input$especie_b,
                   bloque = "enfocada", fuente = "Crossref", query_origen = e$query_origen,
                   cita = e$citas, stringsAsFactors = FALSE)
      } else NULL

      df <- bind_rows(filas_generales, filas_enfocadas)
      write.csv(df, file, row.names = FALSE)
    }
  )
}

shinyApp(ui, server)
