Interacciones Bióticas

Ficha FAIR/CARE de referencias sobre interacción entre especies (WebAssembly)

Interacción entre especies: la co-existencia de especies en un espacio y tiempo brinda un marco regional para la ocurrencia de interacciones entre especies, incluyéndonos.

Distribución geográfica de especies: la co-existencia de especies en un espacio y tiempo brinda un marco regional para la ocurrencia de interacciones entre especies, incluyéndonos. La co-existencia regional de especies forma un marco de co-evolución entre especies, o mosaico geográfico de co-evolución, en donde las especies evolucionan en función de las características de la interacción en el paisaje. La búsqueda de información dirigida a una interacción específica entre especies puede ser sistematizada. Aquí realizamos un ejercicio disparador para empezar a investigar sobre lo conocido de una interacción, con base en las referencias disponibles.

Nota de desarrollo: el resultado que esperamos, como una “micro-síntesis de referencias sobre una interacción entre dos especies”, necesita refinamiento para el filtrado y mejorar la capacidad de búsqueda para enfocarse en la interacción entre las especies seleccionadas.

⏳ Primera carga: el entorno estadístico (R) se descarga y arranca en el navegador — puede tardar un par de minutos la primera vez.

Tip💡 Cómo usar esta ficha
  1. Escribe el nombre científico de la Especie A y la Especie B, y verifica con el botón de chequeo.
  2. Elige 10 rasgos de historia de vida disponibles para cada especie — son metadatos descriptivos (marco: Stearns 1992, Arnold 1983), no determinan un resultado automático.
  3. Aprieta “Mapear Interacción”: la app busca en Crossref citas que mencionen a ambas especies, priorizando trabajos sobre interacciones bióticas. Si no hay suficientes con ese criterio estricto, amplía gradualmente — pero nunca por debajo de “al menos una de las dos especies mencionada”.
  4. Descarga las citas en .csv para el manejo bibliográfico sobre la interacción entre especies seleccionada.

generar_traits_ui <- function(sufijo) {

    def <- traits_def[[id]]

# Reutilizado de bio-distribution.qmd: valida nombres contra el backbone de GBIF
gbif_name_match <- function(name) {
  url <- paste0("https://api.gbif.org/v1/species/match?name=", URLencode(name))
  tryCatch(fromJSON(url), error = function(e) NULL)
}

# mailto: identidad de contacto del proyecto (no personal), reutilizable en OSF/STAPLE
crossref_url <- function(query, rows = 15) {
  paste0("https://api.crossref.org/works?query=", URLencode(query),
         "&rows=", rows, "&mailto=interdomestic@proton.me")
}

# Motor de unión: ¿el título/abstract de cada item menciona a las especies?
especies_en_texto <- function(items, especie_a, especie_b) {
  if (is.null(items) || nrow(items) == 0) {
    return(list(ambas = logical(0), alguna = logical(0)))
  }
  texto <- vapply(seq_len(nrow(items)), function(i) {
    titulo <- tryCatch(paste(items$title[[i]], collapse = " "), error = function(e) "")
    abst <- tryCatch(items$abstract[i], error = function(e) NA)
    abst <- if (is.null(abst) || is.na(abst)) "" else abst
    tolower(paste(titulo, abst))
  }, character(1))
  menciona_a <- grepl(tolower(especie_a), texto, fixed = TRUE)
  menciona_b <- grepl(tolower(especie_b), texto, fixed = TRUE)
  list(ambas = menciona_a & menciona_b, alguna = menciona_a | menciona_b)
}

crossref_buscar <- function(query, rows = 15) {
  tryCatch(fromJSON(crossref_url(query, rows))$message$items, error = function(e) NULL)
}

# Motor de unión de 3 niveles: nunca devuelve citas que no mencionen a ninguna especie
crossref_buscar_citas <- function(especie_a, especie_b, rows = 15) {
  q_interaccion <- paste(especie_a, especie_b,
                          "interaction OR predation OR competition OR mutualism OR parasitism OR facilitation")
  q_simple <- paste(especie_a, especie_b)

  # Nivel 1: interacción + ambas especies mencionadas
  items1 <- crossref_buscar(q_interaccion, rows)
  f1 <- especies_en_texto(items1, especie_a, especie_b)
  if (!is.null(items1) && sum(f1$ambas) >= 3) {
    return(list(items = items1[f1$ambas, , drop = FALSE],
                nivel = "interacciones bióticas, ambas especies mencionadas", consulta = q_interaccion))
  }

  # Nivel 2: sin palabras de interacción, ambas especies mencionadas
  items2 <- crossref_buscar(q_simple, rows)
  f2 <- especies_en_texto(items2, especie_a, especie_b)
  if (!is.null(items2) && sum(f2$ambas) >= 1) {
    return(list(items = items2[f2$ambas, , drop = FALSE],
                nivel = "general, ambas especies mencionadas juntas", consulta = q_simple))
  }

  # Nivel 3: última red — al menos una especie mencionada, nunca ruido total
  base_items <- if (!is.null(items1) && sum(f1$alguna) > 0) items1 else items2
  base_f <- if (!is.null(items1) && sum(f1$alguna) > 0) f1 else f2
  if (!is.null(base_items) && sum(base_f$alguna) >= 1) {
    return(list(items = base_items[base_f$alguna, , drop = FALSE],
                nivel = "baja confianza: mención de al menos una especie", consulta = q_simple))
  }

  list(items = NULL, nivel = "sin resultados que mencionen a ninguna de las dos especies", consulta = q_simple)
}

# Formatea cada item de Crossref como una cita legible
formatear_citas <- function(items) {
  if (is.null(items) || nrow(items) == 0) return(character(0))
  vapply(seq_len(nrow(items)), function(i) {
    tryCatch({
      titulo <- items$title[[i]]
      titulo <- if (length(titulo) > 0) titulo[1] else "Sin título"

      au <- items$author[[i]]
      autores <- if (is.null(au) || nrow(au) == 0) {
        "Autor/es no disponible/s"
      } else {
        nombres <- paste(au$family, au$given, sep = ", ")
        if (length(nombres) > 3) paste0(paste(nombres[1:3], collapse = "; "), " et al.")
        else paste(nombres, collapse = "; ")
      }

      anio <- tryCatch(items$published[[i]]$`date-parts`[[1]][1], error = function(e) NA)
      revista <- tryCatch(items$`container-title`[[i]][1], error = function(e) NA)
      doi <- items$DOI[i]
      doi_url <- if (!is.null(doi) && !is.na(doi)) paste0("https://doi.org/", doi) else NA

      paste0(
        autores, " (", ifelse(is.na(anio), "s.f.", anio), "). ", titulo,
        ifelse(!is.na(revista), paste0(". ", revista, "."), "."),
        ifelse(!is.na(doi_url), paste0(" ", doi_url), "")
      )
    }, error = function(e) "(No se pudo formatear esta cita.)")
  }, character(1))
}

# ==========================================
# SERVIDOR
# ==========================================
server <- function(input, output, session) {

  validacion <- eventReactive(input$verificar, {
    list(
      a = list(consulta = input$especie_a, match = gbif_name_match(input$especie_a)),
      b = list(consulta = input$especie_b, match = gbif_name_match(input$especie_b))
    )
  })

  output$validacion_nombres <- renderUI({
    v <- validacion()
    tagList(lapply(list(v$a, v$b), function(r) {
      m <- r$match
      if (is.null(m) || is.null(m$matchType) || m$matchType == "NONE") {
        div(class = "alert alert-danger", style = "padding:6px; font-size:0.85em; margin-top:4px;",
            paste0("❌ '", r$consulta, "' no se encontró en el backbone de GBIF."))
      } else if (m$matchType == "EXACT") {
        div(class = "alert alert-success", style = "padding:6px; font-size:0.85em; margin-top:4px;",
            paste0("✅ '", r$consulta, "' reconocido como ", m$scientificName, " (", m$status, ")"))
      } else {
        div(class = "alert alert-warning", style = "padding:6px; font-size:0.85em; margin-top:4px;",
            paste0("⚠️ '", r$consulta, "' es una coincidencia aproximada. GBIF sugiere: ", m$canonicalName, "."))
      }
    }))
  })

  resumen_rasgos <- function(sufijo) {
    paste(sapply(names(traits_def), function(id) {
      paste0(traits_def[[id]]$label, ": ", input[[paste0(id, "_", sufijo)]])
    }), collapse = " · ")
  }

  resultado <- eventReactive(input$procesar, {
    req(input$especie_a, input$especie_b)
    withProgress(message = "Buscando citas en Crossref (motor de unión de 3 niveles)...", value = 0.3, {
      res <- crossref_buscar_citas(input$especie_a, input$especie_b)
      incProgress(0.7)
      res
    })
  })

  output$ficha_resultado <- renderUI({
    res <- resultado()
    req(res)
    citas <- formatear_citas(res$items)

    wellPanel(
      style = "background: #ffffff; border: 1px solid #d1d5db;",
      tags$h4("Ficha de Interacción"),
      tags$p(tags$strong("Especie A: "), input$especie_a),
      tags$p(style = "font-size: 0.85em; color: #64748b;", resumen_rasgos("a")),
      tags$p(tags$strong("Especie B: "), input$especie_b),
      tags$p(style = "font-size: 0.85em; color: #64748b;", resumen_rasgos("b")),
      tags$hr(),
      tags$pre(
        paste0(
          "--- METADATA PROFILE (FAIR/CARE) ---\n",
          "Fuente de citas: Crossref REST API\n",
          "Consulta ejecutada: ", res$consulta, "\n",
          "Nivel de resultados (motor de unión): ", res$nivel, "\n",
          "Citas encontradas: ", length(citas), "\n",
          "Fecha de consulta: ", format(Sys.time(), "%Y-%m-%d %H:%M %Z")
        )
      ),
      tags$hr(),
      if (length(citas) == 0) {
        tags$p("No se encontraron citas que mencionen a ninguna de las dos especies.")
      } else {
        tags$ol(lapply(citas, function(c) tags$li(HTML(c))))
      },
      br(),
      downloadButton("descargar_citas", "Descargar citas (.csv)", class = "btn-outline-secondary")
    )
  })

  output$descargar_citas <- downloadHandler(
    filename = function() paste0("interdomestic_citas_", Sys.Date(), ".csv"),
    content = function(file) {
      res <- resultado()
      req(res)
      citas <- formatear_citas(res$items)
      df <- data.frame(
        especie_a = input$especie_a,
        especie_b = input$especie_b,
        nivel_busqueda = res$nivel,
        cita = citas,
        stringsAsFactors = FALSE
      )
      write.csv(df, file, row.names = FALSE)
    }
  )
}
